\section{Introduction}

\subsection{Why non-invasive brain stimulation methods?}
\begin{frame}
\frametitle{Why non-invasive deep brain stimulation methods?}
    \begin{itemize}
        \uncover<1->{\item Invasive methods pose great risk for the patient}
        \uncover<2->{\item Minimum recovery time, with easier and more scalable testing}
        \uncover<3->{\item Used in the treatment of depression (tDCS) \cite{Moffa2020,Jog2019} and other conditions \cite{Cagnan2019}}
        \uncover<4->{\item Most of key structures lie deep within the brain}
    \end{itemize}
\end{frame}

\subsection{Transcranial Electric Stimulation (tES)}
\begin{frame}
\frametitle{Transcranial Electric Stimulation (tES)}

\begin{columns}
    \begin{column}{0.5\textwidth}

\subsection{A New Method (tTIS)}
\begin{frame}
\frametitle{A New Method (tTIS)}

\begin{columns}
    \begin{column}{0.5\textwidth}
        \begin{itemize}
            \uncover<1->{\item Introduced by Grossman et al. in 2017 \cite{Grossman2017}}
            \uncover<2->{\item Tested on murine subjects and was successful \cite{Grossman2017}}
            \uncover<3->{\item Achieves great focality and penetration depth}
            \uncover<4->{\item Some simulation done on human models \cite{Rampersad2019}}
